#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int a, b, c;
    cin >> a >> b >> c;

    // نطبع الأرقام مرتبة
    int arr[3] = {a, b, c};
    sort(arr, arr + 3);
    for (int i = 0; i < 3; i++)
        cout << arr[i] << endl;

    cout << endl;  // سطر فاضي

    // نطبع الأرقام الأصلية
    cout << a << " " << b << " " << c << endl;

    return 0;
}
